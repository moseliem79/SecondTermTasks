/* Write a java program that checks if a set of 10 number enetered by user are in the ascending order (in this case print congrats message) or not (in this case print the number of conflicts)
 */
import java.util.Scanner;
public class Main
{
 public static void main(String[] args) {
 Scanner input = new Scanner(System.in);
 int conflicts = 0;
 int latestValue = 0;
 for (int i = 0; i < 10; i++) {
 System.out.print("Enter the " + (i+1) + "th number: ");
 int n = input.nextInt();
 // Check if the numbers are in order so far
 if (i > 0 && n < latestValue) {
           // if it conflict print why?
           System.out.println("Conflict: " + n + " < " + latestValue );
           conflicts++;
 }
 latestValue=n; 
 }
 // Print the result
 if (conflicts == 0) {
 System.out.println("Mohammed Seleim say! Congrats. All number are ordered.");
 } else {
 System.out.println("Mohammed Seleim say! Unfortunately, you have " + conflicts + " conflicts.");
 }
 }
}
